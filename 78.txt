        Сделайте так, чтобы каждый последующий промис через 3 секунды возводил в квадрат
        результат предыдущего промиса. После окончания манипуляций выведите число алертом на экран.</h3>

    <script>

        const promise3 = new Promise((resolve, reject) => {

            const num = getRandomInt(1, 10);

            console.log('1', num);

            setTimeout(() => {

                resolve(num);

            }, 3000);

        });



        promise3
            .then(

                num => new Promise((resolve, reject) => {

                    console.log('2', num * 2);

                    setTimeout(() => {

                        resolve(num * 2);

                    }, 3000);

                })

            )
            .then(

                num => new Promise((resolve, reject) => {

                    console.log('3', num * 3);

                    setTimeout(() => {

                        resolve(num * 3);

                    }, 3000);

                })

            )
            .then(

                num => console.log('end', num)
                

            );




        function getRandomInt(min, max) {

            return Math.floor(Math.random() * (max - min + 1)) + min;

        };



    </script>
